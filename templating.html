Basic partial

<p>Include me!</p>


Send text to a partial

<p>
  Welcome back, <%= name %>.
</p>


Loop inside a partial

<p>
  <% for (i = 1; i == 10; i++) { %>
    <%= i %>
  <% } %>
</p>


Using conditionals

<p>
  <% if (logged_in) { %>
    My account
  <% } else { %>
    Log out
  <% } %>
</p>


Send an array to a partial

<ul>
  <% offices.split(',').each(function(officeName) { %>
    <li>
      <%= officeName %>
    </li>
  <% )}; %>
</ul>


Include external data

<ul>
  <% arrayForPartial.forEach(function(officeName) { %>
    <li>
      <%= officeName %>
    </li>
  <% )}; %>
</ul>


Advanced templating with JS

<p>
  <% var date = new Date(); %>
  Current date:
  <%= date.getMonth() + 1 %>/<%= date.getDay() %>/<%= date.getFullYear() %>
</p>
<p>
  Window width at pageload:
  <%= $(window).width() %>px
</p>
<p>
  Body font:
  <%= $('body').css('font-family') %>
</p>
